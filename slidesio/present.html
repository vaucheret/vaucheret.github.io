Call epresent-run to present an Org-mode buffer. 

;;;###autoload
(defun epresent-run ()
  (interactive)
  (unless (eq major-mode 'org-mode)
    (error "EPresent can only be used from Org Mode"))
  (setq epresent--org-buffer (current-buffer))
  (epresent--get-frame)
  (epresent-mode)
  (epresent-top))
